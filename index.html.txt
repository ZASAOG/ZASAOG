function applyLang(){
let d=data[lang];

["navHome","navRules","navGoals","navJoin","navContact"]
.forEach((id,i)=>document.getElementById(id).innerText=d.nav[i]);

document.getElementById("homeTitle").innerText=d.homeTitle;
document.getElementById("homeDesc").innerText=d.homeDesc;

document.getElementById("rulesTitle").innerText=d.rulesTitle;
document.getElementById("rulesList").innerHTML=d.rules.map(x=>"<li>"+x+"</li>").join("");

document.getElementById("joinText").innerText=d.joinText;

document.getElementById("contactTitle").innerText=d.contactTitle;
}

function toggleLang(){
lang = lang==="sw" ? "en":"sw";
document.querySelector(".lang").innerText = lang==="sw" ? "EN":"SW";
applyLang();
}

function show(id){
document.querySelectorAll(".section").forEach(s=>s.classList.remove("active"));
document.getElementById(id).classList.add("active");
}
