<script>

// LOADER
window.onload = ()=> document.getElementById("loader").style.display="none";

// SIDEBAR
function toggleSidebar(){
  let s = document.getElementById("sidebar");
  s.style.left = s.style.left === "0px" ? "-250px" : "0px";
}

// SECTION SWITCH
function show(id){
  document.querySelectorAll('.card').forEach(c=>c.style.display="none");
  document.getElementById(id).style.display="block";
}

// LANGUAGE
let lang="sw";
function toggleLang(){
  lang = lang==="sw"?"en":"sw";
  document.getElementById("maint").innerText =
    lang==="sw"?"Huduma ipo kwenye matengenezo":"Service under maintenance";
}

// CHATBOX
function toggleChat(){
  let c=document.getElementById("chatbox");
  c.style.display=c.style.display==="block"?"none":"block";
}

// AI BOT (SMART LOCAL)
function getAIResponse(input){
  input = input.toLowerCase();

  if(input.includes("zasa") || input.includes("what is zasa")){
    return "ZASA ni Zanzibar Aerospace Students Association, taasisi ya wanafunzi wa KIST.";
  }
  if(input.includes("malengo") || input.includes("goals")){
    return "Malengo ni kukuza elimu ya aerospace na kuunganisha wanafunzi.";
  }
  if(input.includes("jiunga") || input.includes("register")){
    return "Pakua fomu kwenye sehemu ya Kujisajili.";
  }
  if(input.includes("contact") || input.includes("mawasiliano")){
    return "Piga simu 0714095979 au WhatsApp.";
  }
  if(input.includes("hello") || input.includes("hi")){
    return "Karibu ZASA! Naweza kukusaidia nini?";
  }

  return "Samahani, tafadhali uliza kuhusu ZASA, malengo au usajili.";
}

function sendMessage(){
  let input=document.getElementById("userInput").value;
  let box=document.getElementById("messages");

  box.innerHTML += "<div><b>You:</b> "+input+"</div>";

  let reply=getAIResponse(input);

  setTimeout(()=>{
    box.innerHTML += "<div><b>AI:</b> "+reply+"</div>";
  },500);
}
