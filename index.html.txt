<script>

/* LOADER */
window.onload = function(){
setTimeout(()=>{
document.getElementById("loader").style.display="none";
},1500);
}

/* MENU */
function toggleMenu(){
document.getElementById("sidebar").classList.toggle("show");
}

/* SECTION */
function showSection(id){
document.querySelectorAll(".section").forEach(sec=>sec.classList.remove("active"));
document.getElementById(id).classList.add("active");
toggleMenu();
}

/* JOIN */
function openJoin(){
document.getElementById("joinMsg").innerText="⚠️ Inafanyiwa kazi";
}

/* FORM */
function downloadForm(){
window.location.href="fomu.pdf";
}

/* KATIBA */
function openKatiba(){
document.getElementById("katibaBox").style.display="block";
}

/* AI */
function toggleAI(){
let box=document.getElementById("aiBox");
box.style.display = box.style.display==="flex" ? "none":"flex";
box.style.flexDirection="column";
}

function sendAI(){
let input=document.getElementById("aiInput");
let chat=document.getElementById("aiChat");

let userMsg=input.value.trim();
if(userMsg==="") return;

chat.innerHTML += "<div><b>Wewe:</b> "+userMsg+"</div>";

let reply="";

let msg=userMsg.toLowerCase();

if(msg.includes("zasa")){
reply="ZASA ni taasisi ya anga kwa vijana.";
}
else if(msg.includes("kujiunga")){
reply="Nenda sehemu ya Join kujiunga.";
}
else if(msg.includes("katiba")){
reply="Katiba ipo kwenye section ya Katiba.";
}
else{
reply="Samahani bado najifunza 🤖";
}

setTimeout(()=>{
chat.innerHTML += "<div><b>AI:</b> "+reply+"</div>";
chat.scrollTop = chat.scrollHeight;
},500);

input.value="";
}
</script>
