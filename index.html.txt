function showSection(sec){
  document.getElementById("content").innerHTML = data[lang][sec];
}

function toggleMenu(){
  let m=document.getElementById("menu");
  m.style.left = (m.style.left=="0px")?"-250px":"0px";
}

function toggleLang(){
  lang = (lang=="sw")?"en":"sw";
  document.querySelector(".lang-btn").innerText = (lang=="sw")?"EN":"SW";
  showSection("home");
}

function toggleChat(){
  let c=document.getElementById("chatBox");
  c.style.display = (c.style.display=="block")?"none":"block";
}

function askAI(e){
  if(e.key==="Enter"){
    let input=e.target.value.toLowerCase();
    let response="Samahani, naweza kusaidia kuhusu ZASA tu.";

    if(input.includes("zasa")) response="ZASA ni chama cha wanafunzi wa aerospace.";
    if(input.includes("join")) response="Tumia sehemu ya Kujiunga kwenye menu.";
    if(input.includes("contact")) response="Piga 0714095979.";

    document.getElementById("chatLog").innerHTML += "<div>> "+input+"</div><div>"+response+"</div>";
    e.target.value="";
  }
}
